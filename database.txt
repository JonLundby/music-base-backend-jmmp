Create Database musicBase_db;
CREATE TABLE artists (
    artistID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name varchar(255),
    birthdate date,
    genres varchar(255),
    website varchar(255),
    image varchar(255),
    numberOfAlbums int
);

CREATE TABLE albums (
    albumID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    albumCover varchar(255),
    releaseDate varchar(255),
    albumTitle varchar(255),
    numberofTracks int
);

CREATE TABLE tracks (
    trackID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    trackName varchar(255),
    duration time
);

CREATE TABLE track_artists (
    trackID int,
    artistID int,
    PRIMARY KEY (trackID, artistID),
    FOREIGN KEY (trackID) REFERENCES tracks(trackID),
    FOREIGN KEY (artistID) REFERENCES artists(artistID)
);

CREATE TABLE track_albums (
    trackID int,
    albumID int,
    PRIMARY KEY (trackID, albumID),
    FOREIGN KEY (trackID) REFERENCES tracks(trackID),
    FOREIGN KEY (albumID) REFERENCES albums(albumID)
);


CREATE TABLE album_artists (
    albumID int,
    artistID int,
    PRIMARY KEY (albumID, artistID),
    FOREIGN KEY (albumID) REFERENCES albums(albumID),
    FOREIGN KEY (artistID) REFERENCES artists(artistID)
);
